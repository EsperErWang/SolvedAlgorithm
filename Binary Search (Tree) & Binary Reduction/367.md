public boolean isPerfectSquare(int num) {
    long left = 0;
    long right = num;
    while (left + 1 < right) {
        long mid = left + (right-left) / 2;
        long midValue = mid * mid;
        if (midValue == num) {
            return true;
        }
        else if (midValue > num) {
            right = mid;
        } else {
            left = mid;
        }
    }

    if (right * right == num || left * left == num) {
        return true;
    }
    return false;
}
