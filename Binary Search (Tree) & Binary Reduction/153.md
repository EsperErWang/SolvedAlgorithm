public int findMin(int[] nums) {
    int first = 0;
    int last = nums.length - 1;

    if (nums.length == 1) {
        return nums[0];
    }

    while (first < last - 1) {
        int mid = first + (last - first) / 2;
        if (nums[mid] < nums[last]) {
            last = mid;                
        }else {
            first = mid;
        }                
    } 

    return nums[first] < nums[last] ? nums[first] : nums[last];       
}
