public int mySqrt(int x) {
    if (x == 0) {
        return 0;
    }else if (x == 1) {
        return 1;
    }

    long left = 1;
    long right = x / 2;
    while (right > left + 1) {
        long mid = left + (right - left) / 2;
        if (mid * mid > x) {
            right = mid;
        }else if (mid * mid < x) {
            left = mid;
        }else {
            return (int)mid;
        }
    }

    return right * right > x ? (int)left : (int)right;
}
