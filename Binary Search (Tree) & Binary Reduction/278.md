// Simple binary search problem
public int firstBadVersion(int n) {
    int first = 0;
    int last = n;
    while (last > first + 1) {
        int mid = first + (last - first) / 2;
        if (isBadVersion(mid)) {
            last = mid;
        }else {
            first = mid;
        }
    }
    return isBadVersion(first) ? first : last;       
}
